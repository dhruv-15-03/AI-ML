2.) Encapsulation - Binding the module and providing the functionality for which user has access.
class Bank:
    def __init__(self,amount):
        self.__balance=amount
    @property
    def getBalance(self):
        return self.__balance
    def deposit(self, amount):
        self.__balance+=amount
    def withdraw(self, amount):
        self.__balance-=amount
    
Dhruv=Bank(2000)
print(Dhruv.getBalance)
Dhruv.deposit(1000)
print(Dhruv.getBalance)
Dhruv.withdraw(2500)
print(Dhruv.getBalance)

3.) Inheritance - Child class inherits the property of a parent class
class Animal:
    def __init__(self,name):
        self.name=name
    def message(self):
        print(f"{self.name} is an animal")

class Dog(Animal):
    def __init__(self,name):
        self.name=name
    def spmessage(self, leg=4):
        print(f"{self.name} is a {leg} leg dog")
        
a=Animal("B")
a.message()
c=Dog('D')
c.message()
c.spmessage(3)
